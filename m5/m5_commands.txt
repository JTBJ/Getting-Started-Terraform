terraform init
terraform workspace new development
terraform plan -var-file="terraform.tfvars" -var-file="Development\development.tfvars" -out development.tfplan
terraform apply "development.tfplan"

terraform workspace new uat
terraform plan -var-file="terraform.tfvars" -var-file="UAT\uat.tfvars" -out uat.tfplan
terraform apply "uat.tfplan"

terraform workspace new production
terraform plan -var-file="terraform.tfvars" -var-file="Production\production.tfvars" -out production.tfplan
#Unless you change the domain to something you have, the CNAME DNS entry will fail
terraform apply "production.tfplan"

terraform destroy -var-file="terraform.tfvars" -var-file="Production\production.tfvars"
